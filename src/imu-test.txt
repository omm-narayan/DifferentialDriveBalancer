/*
 * ICM-20948 TILT → MOTOR DIRECTION (DEBUG VERSION)
 * ICM address: 0x69 (ADA → 3.3V)
 * X slot → Motor 1 | Z slot → Motor 2
 */

#include <Arduino.h>
#include <Wire.h>

#define X_STEP_PIN   2
#define X_DIR_PIN    5
#define Z_STEP_PIN   4
#define Z_DIR_PIN    7
#define EN_PIN       8

#define ICM_ADDR     0x69
#define ICM_BANK_SEL 0x7F
#define TILT_THRESHOLD  10.0f
#define STEP_DELAY_US   500

void icmBank(uint8_t b) {
    Wire.beginTransmission(ICM_ADDR);
    Wire.write(ICM_BANK_SEL);
    Wire.write((b & 0x03) << 4);
    Wire.endTransmission();
}

void icmWrite(uint8_t bank, uint8_t reg, uint8_t val) {
    icmBank(bank);
    Wire.beginTransmission(ICM_ADDR);
    Wire.write(reg);
    Wire.write(val);
    Wire.endTransmission();
}

uint8_t icmRead(uint8_t bank, uint8_t reg) {
    icmBank(bank);
    Wire.beginTransmission(ICM_ADDR);
    Wire.write(reg);
    Wire.endTransmission(false);
    Wire.requestFrom((uint8_t)ICM_ADDR, (uint8_t)1);
    return Wire.available() ? Wire.read() : 0xFF;
}

float readTilt() {
    icmBank(0);
    Wire.beginTransmission(ICM_ADDR);
    Wire.write(0x2D); // ACCEL_XOUT_H
    Wire.endTransmission(false);
    Wire.requestFrom((uint8_t)ICM_ADDR, (uint8_t)6);

    uint8_t buf[6] = {0};
    for (int i = 0; i < 6 && Wire.available(); i++) buf[i] = Wire.read();

    float ay = (int16_t)((buf[2] << 8) | buf[3]) / 16384.0f;
    float az = (int16_t)((buf[4] << 8) | buf[5]) / 16384.0f;
    return atan2f(ay, az) * 180.0f / PI;
}

void stepMotors() {
    digitalWrite(X_STEP_PIN, HIGH);
    digitalWrite(Z_STEP_PIN, HIGH);
    delayMicroseconds(2);
    digitalWrite(X_STEP_PIN, LOW);
    digitalWrite(Z_STEP_PIN, LOW);
    delayMicroseconds(STEP_DELAY_US);
}

void setup() {
    Serial.begin(115200);
    delay(1000);
    Serial.println("=== BOOT ===");

    // Motors first
    pinMode(X_STEP_PIN, OUTPUT); digitalWrite(X_STEP_PIN, LOW);
    pinMode(X_DIR_PIN,  OUTPUT); digitalWrite(X_DIR_PIN,  LOW);
    pinMode(Z_STEP_PIN, OUTPUT); digitalWrite(Z_STEP_PIN, LOW);
    pinMode(Z_DIR_PIN,  OUTPUT); digitalWrite(Z_DIR_PIN,  LOW);
    pinMode(EN_PIN,     OUTPUT); digitalWrite(EN_PIN,     LOW); // enable drivers
    Serial.println("Motors enabled");

    // I2C
    Wire.begin();
    Wire.setClock(100000);
    delay(200);

    // Check WHO_AM_I — must return 0xEA for ICM-20948
    uint8_t who = icmRead(0, 0x00);
    Serial.print("WHO_AM_I: 0x");
    Serial.println(who, HEX);

    if (who != 0xEA) {
        Serial.println("ERROR: ICM-20948 not found! Check wiring.");
        Serial.println("Expected 0xEA, got something else.");
        // blink onboard LED to signal error
        pinMode(LED_BUILTIN, OUTPUT);
        while (true) {
            digitalWrite(LED_BUILTIN, HIGH); delay(100);
            digitalWrite(LED_BUILTIN, LOW);  delay(100);
        }
    }

    Serial.println("ICM-20948 found!");

    // Wake up ICM
    icmWrite(0, 0x06, 0x01); delay(50);  // PWR_MGMT_1: wake, auto clock
    icmWrite(0, 0x07, 0x00); delay(20);  // PWR_MGMT_2: enable accel+gyro

    Serial.println("ICM ready. Tilting now...");
}

void loop() {
    float tilt = readTilt();

    // Print tilt every 200ms so you can see it
    static unsigned long lastPrint = 0;
    if (millis() - lastPrint > 200) {
        Serial.print("Tilt: ");
        Serial.println(tilt);
        lastPrint = millis();
    }

    if (tilt > TILT_THRESHOLD) {
        Serial.println(">> FORWARD");
        digitalWrite(X_DIR_PIN, HIGH);
        digitalWrite(Z_DIR_PIN, LOW);
        stepMotors();
    }
    else if (tilt < -TILT_THRESHOLD) {
        Serial.println(">> BACKWARD");
        digitalWrite(X_DIR_PIN, LOW);
        digitalWrite(Z_DIR_PIN, HIGH);
        stepMotors();
    }
    // else flat — do nothing, motors hold position
}
