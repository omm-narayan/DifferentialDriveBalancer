/*
 * NEMA 17 x2 + A4988 + CNC Shield v3 — Two Motor Back & Forth
 * Board  : NUCLEO-F446RE (STM32duino / PlatformIO)
 * MS1+MS2+MS3 all shorted → 1/16 microstepping → 3200 steps/rev
 *
 * Motor 1 → X slot on CNC shield  (no changes)
 * Motor 2 → Z slot on CNC shield  (use Z instead of Y — avoids D3/PB3 conflict)
 *
 * Pin assignments:
 *   X STEP → D2  (PA10)   X DIR → D5 (PB4)
 *   Z STEP → D4  (PB5)    Z DIR → D7 (PA8)
 *   EN     → D8  (PA9)    shared, active LOW
 *
 * No jumper wires needed — plug motors into X and Z slots directly.
 */

#include <Arduino.h>

#define X_STEP_PIN      2   // D2  → PA10
#define X_DIR_PIN       5   // D5  → PB4
#define Z_STEP_PIN      4   // D4  → PB5
#define Z_DIR_PIN       7   // D7  → PA8
#define EN_PIN          8   // D8  → PA9 (shared, LOW = enabled)

#define STEPS_PER_REV   3200
#define REVOLUTIONS     2
#define STEP_DELAY_US   150

const long TOTAL_STEPS = (long)STEPS_PER_REV * REVOLUTIONS;

void doStep()
{
    digitalWrite(X_STEP_PIN, HIGH);
    digitalWrite(Z_STEP_PIN, HIGH);
    delayMicroseconds(2);
    digitalWrite(X_STEP_PIN, LOW);
    digitalWrite(Z_STEP_PIN, LOW);
    delayMicroseconds(STEP_DELAY_US);
}

void moveSteps(long steps)
{
    for (long i = 0; i < steps; i++) doStep();
}

void setup()
{
    pinMode(X_STEP_PIN, OUTPUT);
    pinMode(X_DIR_PIN,  OUTPUT);
    pinMode(Z_STEP_PIN, OUTPUT);
    pinMode(Z_DIR_PIN,  OUTPUT);
    pinMode(EN_PIN,     OUTPUT);

    digitalWrite(X_STEP_PIN, LOW);
    digitalWrite(X_DIR_PIN,  LOW);
    digitalWrite(Z_STEP_PIN, LOW);
    digitalWrite(Z_DIR_PIN,  LOW);
    digitalWrite(EN_PIN,     LOW);

    delay(100);
}

void loop()
{
    // Both forward
    digitalWrite(X_DIR_PIN, HIGH);
    digitalWrite(Z_DIR_PIN, HIGH);
    delayMicroseconds(5);
    moveSteps(TOTAL_STEPS);
    delay(500);

    // Both reverse
    digitalWrite(X_DIR_PIN, LOW);
    digitalWrite(Z_DIR_PIN, LOW);
    delayMicroseconds(5);
    moveSteps(TOTAL_STEPS);
    delay(500);
}
